async function checkStudentStatus() { const studentId = document.getElementById('studentIdInput').value; const resultBox = document.getElementById('resultBox'); if (!studentId) { alert('Please enter your Student ID'); return; } // Show loading message resultBox.style.display = 'block'; resultBox.className = 'result'; resultBox.innerHTML = '

⏳ Checking your status...

'; try { const response = await fetch(LAMBDA_URL, { method: 'POST', headers: { 'Content-Type': 'application/json', 'Accept': 'application/json' }, body: JSON.stringify({ student_id: studentId }) }); const data = await response.json(); console.log('Response data:', data); if (data.statusCode === 200) { const info = JSON.parse(data.body); resultBox.className = 'result success'; resultBox.innerHTML = `

✅ ${info.message}

Name: ${info.student_name}

School: ${info.school_id}

Semester: ${info.semester}

Access Expires: ${info.expiry_date}

Status: ${info.status}

`; } else { const info = JSON.parse(data.body); resultBox.className = 'result error'; resultBox.innerHTML = `

❌ ${info.message}

${info.details || ''}

`; } } catch (error) { console.error('Error:', error); resultBox.className = 'result error'; resultBox.innerHTML = `

Connection Error

Could not connect to the server.

Error: ${error.message}

Check that your Lambda URL is correct and the function has CORS enabled.

`; } }
